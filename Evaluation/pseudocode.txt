Σύγκριση υπότιτλου Α με Β
(Έχουμε τα timestamps του υπότιτλου Α και του υπότιτλου Β)


Αρχικά: score = 0

sub1 = max_length(A,B) // Πιο πολλές γραμμές υποτίτλων
sub2 = min_length(A,B) // Πιο λίγες γραμμές υποτίτλων

Δημιουγία 4 Διανυσμάτων

sub1_t_start = διάνυσμα που περιέχει όλες τις χρονικές τιμές σε ms που ξεκινάει ο υπότιτλος sub1 (μήκος Ν - ταξινομημένος)
sub1_t_stop = διάνυσμα που περιέχει όλες τις χρονικές τιμές σε ms που τελειώνει ο υπότιτλος sub1 (μήκος Ν - ταξινομημένος)
sub2_t_start = διάνυσμα που περιέχει όλες τις χρονικές τιμές σε ms που ξεκινάει ο υπότιτλος sub2 (μήκος Μ - ταξινομημένος)
sub2_t_stop = διάνυσμα που περιέχει όλες τις χρονικές τιμές σε ms που τελειώνει ο υπότιτλος sub2 (μήκος Μ - ταξινομημένος)

threshold = 500 ms
j = 0
k = 0
Για i = 1 έως Ν (αυτού με τις πολλές):
	
	startFlag = true
	stopFlag  = true
	
	Όσο sub2_t_start(j) < sub1_t_start(i) - threshold
		j = j + 1
	Τέλος_Επανάληψης
	Αν sub2_t_start(j) > sub1_t_start(i) + threshold
		startFlag = false
		j = j - 1
	Τελος_Αν
	
	Όσο sub2_t_stop(k) < sub1_t_stop(i) - threshold
		k = k + 1
	Τέλος_Επανάληψης
	Αν sub2_t_stop(k) > sub1_t_stop(i) + threshold
		stopFlag = false
		k = k - 1
	Τελος_Αν
		
	Αν startFlag == true && stopFlag == true 
		score = score + 1
	Αλλιώς
		score = score - 1
	Τελος_Αν

Τέλος_Επανάληψης

// Το score εχει τιμη στο διάστημα [-Ν,Ν] οποτε το κανονικοποιούμε - [0,1] ως:
